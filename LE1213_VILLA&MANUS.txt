#include <stdio.h>

int main() {
    FILE *fp;
    int n, i;
    char name[50], age[10], course[50];

    // ------------------ WRITING TO FILE ------------------ //
    printf("Writing Student Database:\n");
    printf("How many students do you want to record? ");
    scanf("%d", &n);
    getchar(); // consume newline left by scanf

    fp = fopen("student_records.txt", "w");
    if (fp == NULL) {
        printf("Error opening file for writing.\n");
        return 1;
    }

    for (i = 1; i <= n; i++) {
        printf("\nStudent %d\n", i);

        printf("Enter Name: ");
        fgets(name, sizeof(name), stdin);

        printf("Enter Age: ");
        fgets(age, sizeof(age), stdin);

        printf("Enter Course: ");
        fgets(course, sizeof(course), stdin);

        // Write to file
        fprintf(fp, "Student %d\n", i);
        fprintf(fp, "Name: %s", name);
        fprintf(fp, "Age: %s", age);
        fprintf(fp, "Course: %s\n", course);
        fprintf(fp, "-------------------------\n");
    }

    fclose(fp);
    printf("\nStudent database successfully saved to file.\n");

    // ------------------ READING FROM FILE ------------------ //
    fp = fopen("student_records.txt", "r");
    if (fp == NULL) {
        printf("Error opening file for reading.\n");
        return 1;
    }

    char line[200];
    printf("\nReading Student Database:\n\n");

    while (fgets(line, sizeof(line), fp) != NULL) {
        printf("%s", line);
    }

    fclose(fp);
    return 0;
}
